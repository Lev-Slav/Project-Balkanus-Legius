#Spanish Civil War#
spanish_civil_war-kingdom_support = {
	enable = {
		has_country_flag = SPR_intervention
		has_global_flag = spanish_civil_war
		has_war = no
		is_subject = no
		country_exists = SPR
	}
	abort = { #stop sending troops; Stop if any of the above conditions are no longer met
		OR = {
			NOT = { has_country_flag = SPR_intervention }
			NOT = { has_global_flag = spanish_civil_war }
			has_war = yes
			is_subject = yes
			NOT = { country_exists = SPR } #Spain must exists
		}

	}
	ai_strategy = {
		type = send_volunteers_desire
		id = "SPR"
		value = 100
	}
	ai_strategy = {
		type = support
		id = "SPR"
		value = 100
	}
	ai_strategy = {
		type = ignore
		id = "CAR"
		value = 100
	}
	ai_strategy = {
		type = ignore
		id = "CNT"
		value = 100
	}
}

#Afghan-Delhi War

	enable = {
		NOT = { tag = DEH }
		is_in_faction_with = DEH
		DEH = {

# Negating Strange Invasion AI Behavior in Civil Wars
usa_civil_war_no_invade = {
	enable = {
		tag = USA
		OR = {
			has_war_with = CSA
			has_war_with = CAL
			has_war_with = TEX
		}
	}
	abort = {
		NOT = {
			has_war_with = CSA
			has_war_with = CAL
			has_war_with = TEX
		}
	}
	ai_strategy = {
		type = invade
		id = "CSA"
		value = -500
	}
	ai_strategy = {
		type = invade
		id = "CAL"
		value = -500
	}
	ai_strategy = {
		type = invade
		id = "TEX"
		value = -500
	}
}
